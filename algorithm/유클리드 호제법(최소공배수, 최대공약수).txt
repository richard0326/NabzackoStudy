#include <string>
#include <vector>

using namespace std;

vector<int> solution(int n, int m) {
    vector<int> answer;

    int times = m * n;

    if(n<m) {
        int temp = n;
        n = m;
        m = temp;
    }

    for(;;) {
        int min = n % m;
        if(min == 0)
            break;

        n = m;
        m = min;
    }

    answer.push_back(m);
    answer.push_back(times/m);

    return answer;
}
